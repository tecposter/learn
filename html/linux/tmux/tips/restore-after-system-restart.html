install tmux plugin: tmux-resurrect https://github.com/tmux-plugins/tmux-resurrect

key buildings

prefix + Ctrl-s - save prefix + Ctrl-r - restore

example

prefix + Ctrl-s

prefix + :kill-server

restart server

prefix + Ctrl-r

in .tmux.conf

`

for vim

set -g @resurrect-strategy-vim 'session'

for neovim

set -g @resurrect-strategy-nvim 'session' `
